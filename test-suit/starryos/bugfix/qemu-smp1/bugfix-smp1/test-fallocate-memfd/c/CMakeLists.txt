cmake_minimum_required(VERSION 3.20)
project(test-fallocate-memfd C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-fallocate-memfd src/main.c)
target_include_directories(test-fallocate-memfd PRIVATE src)
target_compile_options(test-fallocate-memfd PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-fallocate-memfd RUNTIME DESTINATION usr/bin)
