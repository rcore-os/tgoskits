cmake_minimum_required(VERSION 3.20)
project(bug-open-dir-wronly C)
set(CMAKE_C_STANDARD 11)

set(CMAKE_C_EXTENSIONS ON)

target_compile_options(bug-open-dir-wronly PRIVATE -Wall -Wextra -Werror)
