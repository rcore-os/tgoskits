cmake_minimum_required(VERSION 3.20)
project(test-netlink-uevent C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-netlink-uevent src/main.c)
target_include_directories(test-netlink-uevent PRIVATE src)
target_compile_options(test-netlink-uevent PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-netlink-uevent RUNTIME DESTINATION usr/bin)
