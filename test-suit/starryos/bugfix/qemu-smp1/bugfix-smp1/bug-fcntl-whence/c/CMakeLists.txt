cmake_minimum_required(VERSION 3.20)
project(bug-fcntl-whence C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS ON)
add_executable(bug-fcntl-whence src/main.c)
target_compile_options(bug-fcntl-whence PRIVATE -Wall -Wextra -Werror)
install(TARGETS bug-fcntl-whence RUNTIME DESTINATION usr/bin)
