cmake_minimum_required(VERSION 3.20)
project(test-netlink-genl C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-netlink-genl src/main.c)
target_include_directories(test-netlink-genl PRIVATE src)
target_compile_options(test-netlink-genl PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-netlink-genl RUNTIME DESTINATION usr/bin)
