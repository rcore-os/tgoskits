cmake_minimum_required(VERSION 3.20)
project(bug-flock C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS ON)
add_executable(bug-flock src/main.c)
target_compile_options(bug-flock PRIVATE -Wall -Wextra -Werror)
install(TARGETS bug-flock RUNTIME DESTINATION usr/bin)
