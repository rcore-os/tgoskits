cmake_minimum_required(VERSION 3.20)
project(test-evdev-minor C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-evdev-minor src/main.c)
target_include_directories(test-evdev-minor PRIVATE src)
target_compile_options(test-evdev-minor PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-evdev-minor RUNTIME DESTINATION usr/bin)
