cmake_minimum_required(VERSION 3.20)
project(test-evdev-prop-bits C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-evdev-prop-bits src/main.c)
target_include_directories(test-evdev-prop-bits PRIVATE src)
target_compile_options(test-evdev-prop-bits PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-evdev-prop-bits RUNTIME DESTINATION usr/bin)
