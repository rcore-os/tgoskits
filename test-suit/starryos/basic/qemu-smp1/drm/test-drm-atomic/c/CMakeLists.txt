cmake_minimum_required(VERSION 3.20)
project(test-drm-atomic C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-drm-atomic src/main.c)
target_include_directories(test-drm-atomic PRIVATE src)
target_compile_options(test-drm-atomic PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-drm-atomic RUNTIME DESTINATION usr/bin)
