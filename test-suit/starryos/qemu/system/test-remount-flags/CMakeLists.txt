cmake_minimum_required(VERSION 3.20)
project(test-remount-flags C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-remount-flags src/main.c)
target_compile_options(test-remount-flags PRIVATE -Wall -Wextra -Werror)
target_link_options(test-remount-flags PRIVATE -static)
install(TARGETS test-remount-flags RUNTIME DESTINATION usr/bin/starry-test-suit)
