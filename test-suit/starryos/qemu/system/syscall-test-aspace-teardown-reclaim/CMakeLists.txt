cmake_minimum_required(VERSION 3.20)
project(test-aspace-teardown-reclaim C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-aspace-teardown-reclaim src/main.c)
target_include_directories(test-aspace-teardown-reclaim PRIVATE src)
target_compile_options(test-aspace-teardown-reclaim PRIVATE -Wall -Wextra -Werror)
target_link_libraries(test-aspace-teardown-reclaim PRIVATE pthread)
install(TARGETS test-aspace-teardown-reclaim RUNTIME DESTINATION usr/bin/starry-test-suit)
