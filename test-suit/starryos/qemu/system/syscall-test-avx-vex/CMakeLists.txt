cmake_minimum_required(VERSION 3.20)
project(test-avx-vex C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-avx-vex src/main.c)
target_include_directories(test-avx-vex PRIVATE src)
target_compile_options(test-avx-vex PRIVATE -Wall -Wextra)
install(TARGETS test-avx-vex RUNTIME DESTINATION usr/bin/starry-test-suit)
