cmake_minimum_required(VERSION 3.20)
project(test-unshare-mount-ns C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-unshare-mount-ns src/main.c)
target_include_directories(test-unshare-mount-ns PRIVATE src)
target_compile_options(test-unshare-mount-ns PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-unshare-mount-ns RUNTIME DESTINATION usr/bin/starry-test-suit)
