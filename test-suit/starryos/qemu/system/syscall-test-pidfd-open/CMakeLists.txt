cmake_minimum_required(VERSION 3.20)
project(test-pidfd-open C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-pidfd-open src/main.c)
target_include_directories(test-pidfd-open PRIVATE src ${CMAKE_CURRENT_SOURCE_DIR}/../common)
target_compile_options(test-pidfd-open PRIVATE -Wall -Wextra -Werror)
target_link_libraries(test-pidfd-open PRIVATE -lpthread)
install(TARGETS test-pidfd-open RUNTIME DESTINATION usr/bin/starry-test-suit)
