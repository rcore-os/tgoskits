cmake_minimum_required(VERSION 3.20)
project(test-procstats C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-procstats src/main.c)
target_include_directories(test-procstats PRIVATE src)
target_compile_options(test-procstats PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-procstats RUNTIME DESTINATION usr/bin/starry-test-suit)
