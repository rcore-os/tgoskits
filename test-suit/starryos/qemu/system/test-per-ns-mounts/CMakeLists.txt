cmake_minimum_required(VERSION 3.20)
project(test-per-ns-mounts C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-per-ns-mounts src/main.c)
target_compile_options(test-per-ns-mounts PRIVATE -Wall -Wextra -Werror)
target_link_options(test-per-ns-mounts PRIVATE -static)
install(TARGETS test-per-ns-mounts RUNTIME DESTINATION usr/bin/starry-test-suit)
