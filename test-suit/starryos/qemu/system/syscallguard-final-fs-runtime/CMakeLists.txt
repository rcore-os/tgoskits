cmake_minimum_required(VERSION 3.20)
project(syscallguard-final-fs-runtime C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS ON)
add_executable(syscallguard-final-fs-runtime src/main.c)
target_include_directories(syscallguard-final-fs-runtime PRIVATE "${CMAKE_CURRENT_LIST_DIR}/../common")
target_compile_options(syscallguard-final-fs-runtime PRIVATE -Wall -Wextra -Werror)
install(TARGETS syscallguard-final-fs-runtime RUNTIME DESTINATION usr/bin/starry-test-suit)
