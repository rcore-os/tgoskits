cmake_minimum_required(VERSION 3.20)
project(test-so-reuseport C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-so-reuseport src/main.c)
target_include_directories(test-so-reuseport PRIVATE src)
target_compile_options(test-so-reuseport PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-so-reuseport RUNTIME DESTINATION usr/bin/starry-test-suit)
