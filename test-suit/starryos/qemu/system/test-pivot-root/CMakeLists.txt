cmake_minimum_required(VERSION 3.20)
project(test-pivot-root C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-pivot-root src/main.c)
target_compile_options(test-pivot-root PRIVATE -Wall -Wextra -Werror)
target_link_options(test-pivot-root PRIVATE -static)
install(TARGETS test-pivot-root RUNTIME DESTINATION usr/bin/starry-test-suit)
