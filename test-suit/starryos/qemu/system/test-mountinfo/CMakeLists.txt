cmake_minimum_required(VERSION 3.20)
project(nix-sandbox-test-mountinfo C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(nix-sandbox-test-mountinfo src/main.c)
target_compile_options(nix-sandbox-test-mountinfo PRIVATE -Wall -Wextra -Werror)
target_link_options(nix-sandbox-test-mountinfo PRIVATE -static)
install(TARGETS nix-sandbox-test-mountinfo RUNTIME DESTINATION usr/bin/starry-test-suit)
