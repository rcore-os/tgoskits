cmake_minimum_required(VERSION 3.20)
project(test-setsockopt C)
include(${CMAKE_CURRENT_LIST_DIR}/../common/starry_arch_filter.cmake)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
starry_arch_filtered_executable(test-setsockopt "x86_64|aarch64|riscv64" "test-setsockopt skipped on unsupported architecture" src/main.c)
target_include_directories(test-setsockopt PRIVATE src)
target_compile_options(test-setsockopt PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-setsockopt RUNTIME DESTINATION usr/bin/starry-test-suit)
