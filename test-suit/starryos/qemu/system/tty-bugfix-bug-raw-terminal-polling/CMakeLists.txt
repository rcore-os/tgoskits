cmake_minimum_required(VERSION 3.20)
project(bug-raw-terminal-polling C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(bug-raw-terminal-polling src/main.c)
target_include_directories(bug-raw-terminal-polling PRIVATE src)
target_compile_options(bug-raw-terminal-polling PRIVATE -Wall -Wextra -Werror)
target_compile_definitions(bug-raw-terminal-polling PRIVATE _POSIX_C_SOURCE=200809L)
install(TARGETS bug-raw-terminal-polling RUNTIME DESTINATION usr/bin/starry-test-suit)
