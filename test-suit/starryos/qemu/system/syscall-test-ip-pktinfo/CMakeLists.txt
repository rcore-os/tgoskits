cmake_minimum_required(VERSION 3.20)
project(test-ip-pktinfo C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-ip-pktinfo src/main.c)
target_include_directories(test-ip-pktinfo PRIVATE src)
target_compile_options(test-ip-pktinfo PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-ip-pktinfo RUNTIME DESTINATION usr/bin/starry-test-suit)
