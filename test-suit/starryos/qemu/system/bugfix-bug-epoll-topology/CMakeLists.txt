cmake_minimum_required(VERSION 3.20)
project(bugfix-bug-epoll-topology C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS ON)
add_executable(bugfix-bug-epoll-topology src/main.c)
target_include_directories(bugfix-bug-epoll-topology PRIVATE "${CMAKE_CURRENT_LIST_DIR}/../common")
target_compile_options(bugfix-bug-epoll-topology PRIVATE -Wall -Wextra -Werror)
target_link_libraries(bugfix-bug-epoll-topology PRIVATE pthread)
install(TARGETS bugfix-bug-epoll-topology RUNTIME DESTINATION usr/bin/starry-test-suit)
