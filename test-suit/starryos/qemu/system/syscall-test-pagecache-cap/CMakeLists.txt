cmake_minimum_required(VERSION 3.20)
project(test-pagecache-cap C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-pagecache-cap src/main.c)
target_include_directories(test-pagecache-cap PRIVATE src)
target_compile_options(test-pagecache-cap PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-pagecache-cap RUNTIME DESTINATION usr/bin/starry-test-suit)
