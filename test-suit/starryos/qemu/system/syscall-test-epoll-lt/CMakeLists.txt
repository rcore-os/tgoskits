cmake_minimum_required(VERSION 3.20)
project(test-epoll-lt C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-epoll-lt src/main.c)
target_include_directories(test-epoll-lt PRIVATE src)
target_compile_options(test-epoll-lt PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-epoll-lt RUNTIME DESTINATION usr/bin/starry-test-suit)
