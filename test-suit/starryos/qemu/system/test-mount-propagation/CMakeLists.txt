cmake_minimum_required(VERSION 3.20)
project(test-mount-propagation C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-mount-propagation src/main.c)
target_compile_options(test-mount-propagation PRIVATE -Wall -Wextra -Werror)
target_link_options(test-mount-propagation PRIVATE -static)
install(TARGETS test-mount-propagation RUNTIME DESTINATION usr/bin/starry-test-suit)
