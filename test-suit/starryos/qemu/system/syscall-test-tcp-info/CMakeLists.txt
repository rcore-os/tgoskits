cmake_minimum_required(VERSION 3.20)
project(test-tcp-info C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-tcp-info src/main.c)
target_include_directories(test-tcp-info PRIVATE src ${CMAKE_CURRENT_SOURCE_DIR}/../common)
target_compile_options(test-tcp-info PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-tcp-info RUNTIME DESTINATION usr/bin/starry-test-suit)
