cmake_minimum_required(VERSION 3.20)
project(test-unshare-fs C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-unshare-fs src/main.c)
target_compile_options(test-unshare-fs PRIVATE -Wall -Wextra -Werror)
target_link_libraries(test-unshare-fs PRIVATE pthread)
install(TARGETS test-unshare-fs RUNTIME DESTINATION usr/bin/starry-test-suit)
