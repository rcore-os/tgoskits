cmake_minimum_required(VERSION 3.20)
project(test-pty-openpt C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-pty-openpt src/main.c)
target_include_directories(test-pty-openpt PRIVATE src)
target_compile_options(test-pty-openpt PRIVATE -Wall -Wextra -Werror)
target_link_options(test-pty-openpt PRIVATE -static)
install(TARGETS test-pty-openpt RUNTIME DESTINATION usr/bin/starry-test-suit)
