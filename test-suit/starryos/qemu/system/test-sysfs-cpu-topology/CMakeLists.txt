cmake_minimum_required(VERSION 3.20)
project(test-sysfs-cpu-topology C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)

target_include_directories(test-sysfs-cpu-topology PRIVATE src)
target_compile_options(test-sysfs-cpu-topology PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-sysfs-cpu-topology RUNTIME DESTINATION usr/bin/starry-test-suit)
