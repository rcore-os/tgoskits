cmake_minimum_required(VERSION 3.20)
project(test-futex-clone-thread C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-futex-clone-thread src/main.c)
target_include_directories(test-futex-clone-thread PRIVATE src)
target_compile_options(test-futex-clone-thread PRIVATE -Wall -Wextra -Werror)
find_package(Threads REQUIRED)
target_link_libraries(test-futex-clone-thread PRIVATE Threads::Threads)
install(TARGETS test-futex-clone-thread RUNTIME DESTINATION usr/bin)
