cmake_minimum_required(VERSION 3.20)
project(test-fsync C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-fsync src/main.c)
target_include_directories(test-fsync PRIVATE src)
target_compile_options(test-fsync PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-fsync RUNTIME DESTINATION usr/bin)
