cmake_minimum_required(VERSION 3.20)
project(test-shutdown C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-shutdown src/main.c)
target_include_directories(test-shutdown PRIVATE src)
target_compile_options(test-shutdown PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-shutdown RUNTIME DESTINATION usr/bin)
