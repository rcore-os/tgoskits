cmake_minimum_required(VERSION 3.20)
project(test-getsockname C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-getsockname src/main.c)
target_include_directories(test-getsockname PRIVATE src)
target_compile_options(test-getsockname PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-getsockname RUNTIME DESTINATION usr/bin)
