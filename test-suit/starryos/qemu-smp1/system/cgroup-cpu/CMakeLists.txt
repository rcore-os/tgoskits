cmake_minimum_required(VERSION 3.20)
project(cgroup-cpu C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)

add_executable(cgroup-cpu src/main.c)
target_include_directories(cgroup-cpu PRIVATE src)
target_compile_options(cgroup-cpu PRIVATE -Wall -Wextra -Werror)
install(TARGETS cgroup-cpu RUNTIME DESTINATION usr/bin/starry-test-suit)
