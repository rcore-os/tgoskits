cmake_minimum_required(VERSION 3.20)
project(dma-heap-dmabuf C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)

add_executable(dma-heap-dmabuf src/main.c)
target_compile_options(dma-heap-dmabuf PRIVATE -Wall -Wextra -Werror)
target_link_options(dma-heap-dmabuf PRIVATE -static)
install(TARGETS dma-heap-dmabuf RUNTIME DESTINATION usr/bin/starry-test-suit)
