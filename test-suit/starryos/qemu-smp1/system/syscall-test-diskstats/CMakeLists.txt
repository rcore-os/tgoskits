cmake_minimum_required(VERSION 3.20)
project(test-diskstats C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-diskstats src/main.c)
target_include_directories(test-diskstats PRIVATE src)
target_compile_options(test-diskstats PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-diskstats RUNTIME DESTINATION usr/bin/starry-test-suit)
