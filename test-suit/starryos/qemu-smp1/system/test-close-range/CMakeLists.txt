cmake_minimum_required(VERSION 3.20)
project(test-close-range C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-close-range src/main.c)
target_include_directories(test-close-range PRIVATE src)
target_compile_options(test-close-range PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-close-range RUNTIME DESTINATION usr/bin/starry-test-suit)
