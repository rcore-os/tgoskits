cmake_minimum_required(VERSION 3.20)
project(starry-system-smp1 C)

file(GLOB STARRY_SYSTEM_SUBCASES
    RELATIVE "${CMAKE_CURRENT_LIST_DIR}"
    CONFIGURE_DEPENDS "*"
)
list(SORT STARRY_SYSTEM_SUBCASES)

if(DEFINED STARRY_GROUPED_C_SUBCASES)
    set(STARRY_SYSTEM_SELECTED_SUBCASES ${STARRY_GROUPED_C_SUBCASES})
else()
    set(STARRY_SYSTEM_SELECTED_SUBCASES ${STARRY_SYSTEM_SUBCASES})
endif()

foreach(STARRY_SYSTEM_SUBCASE IN LISTS STARRY_SYSTEM_SELECTED_SUBCASES)
    set(STARRY_SYSTEM_SUBCASE_DIR "${CMAKE_CURRENT_LIST_DIR}/${STARRY_SYSTEM_SUBCASE}")
    if(IS_DIRECTORY "${STARRY_SYSTEM_SUBCASE_DIR}"
        AND EXISTS "${STARRY_SYSTEM_SUBCASE_DIR}/CMakeLists.txt")
        add_subdirectory("${STARRY_SYSTEM_SUBCASE}")
    elseif(DEFINED STARRY_GROUPED_C_SUBCASES)
        message(FATAL_ERROR "unknown grouped C subcase: ${STARRY_SYSTEM_SUBCASE}")
    endif()
endforeach()
