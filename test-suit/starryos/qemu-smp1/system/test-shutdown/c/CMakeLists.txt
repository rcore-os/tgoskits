cmake_minimum_required(VERSION 3.20)
project(test-shutdown C)
include(${CMAKE_CURRENT_LIST_DIR}/../../common/starry_arch_filter.cmake)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
starry_arch_filtered_executable(test-shutdown "x86_64|aarch64|riscv64" "test-shutdown skipped on unsupported architecture" src/main.c)
target_include_directories(test-shutdown PRIVATE src)
target_compile_options(test-shutdown PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-shutdown RUNTIME DESTINATION usr/bin/starry-test-suit)
