cmake_minimum_required(VERSION 3.20)
project(test-getrusage C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-getrusage src/main.c)
target_include_directories(test-getrusage PRIVATE src)
target_compile_options(test-getrusage PRIVATE -Wall -Wextra)
install(TARGETS test-getrusage RUNTIME DESTINATION usr/bin/starry-test-suit)
