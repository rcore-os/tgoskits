cmake_minimum_required(VERSION 3.20)
project(rga-blit-hw C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)

add_executable(rga-blit-hw src/main.c)
target_compile_options(rga-blit-hw PRIVATE -Wall -Wextra -Werror)
target_link_options(rga-blit-hw PRIVATE -static)
install(TARGETS rga-blit-hw RUNTIME DESTINATION usr/bin/starry-test-suit)
