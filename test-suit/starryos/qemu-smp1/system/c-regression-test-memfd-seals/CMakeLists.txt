cmake_minimum_required(VERSION 3.20)
project(test-memfd-seals C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-memfd-seals src/main.c)
target_include_directories(test-memfd-seals PRIVATE src)
target_compile_options(test-memfd-seals PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-memfd-seals RUNTIME DESTINATION usr/bin/starry-test-suit)
