cmake_minimum_required(VERSION 3.20)
project(test-proc-wchan C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-proc-wchan src/main.c)
target_include_directories(test-proc-wchan PRIVATE src ${CMAKE_CURRENT_SOURCE_DIR}/../common)
target_compile_options(test-proc-wchan PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-proc-wchan RUNTIME DESTINATION usr/bin/starry-test-suit)
