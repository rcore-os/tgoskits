cmake_minimum_required(VERSION 3.20)
project(test-xattr C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-xattr src/main.c)
target_include_directories(test-xattr PRIVATE src)
target_compile_options(test-xattr PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-xattr RUNTIME DESTINATION usr/bin/starry-test-suit)
