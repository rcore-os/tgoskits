cmake_minimum_required(VERSION 3.20)
project(test-getsockopt C)
include(${CMAKE_CURRENT_LIST_DIR}/../common/starry_arch_filter.cmake)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
starry_arch_filtered_executable(test-getsockopt "x86_64|aarch64|riscv64" "test-getsockopt skipped on unsupported architecture" src/main.c)
target_include_directories(test-getsockopt PRIVATE src)
target_compile_options(test-getsockopt PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-getsockopt RUNTIME DESTINATION usr/bin/starry-test-suit)
