cmake_minimum_required(VERSION 3.20)
project(test-dup2 C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-dup2 src/main.c)
target_include_directories(test-dup2 PRIVATE src)
target_compile_options(test-dup2 PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-dup2 RUNTIME DESTINATION usr/bin/starry-test-suit)
install(PROGRAMS profile.d/99-test-dup2-autorun.sh DESTINATION etc/profile.d)
