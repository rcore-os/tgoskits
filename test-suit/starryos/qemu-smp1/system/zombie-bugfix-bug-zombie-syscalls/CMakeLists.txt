cmake_minimum_required(VERSION 3.20)
project(bug-zombie-syscalls C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(bug-zombie-syscalls src/main.c)
target_include_directories(bug-zombie-syscalls PRIVATE src)
target_compile_options(bug-zombie-syscalls PRIVATE -Wall -Wextra -Werror)
target_compile_definitions(bug-zombie-syscalls PRIVATE _POSIX_C_SOURCE=200809L)
install(TARGETS bug-zombie-syscalls RUNTIME DESTINATION usr/bin/starry-test-suit)
