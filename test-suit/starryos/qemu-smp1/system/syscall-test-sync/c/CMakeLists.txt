cmake_minimum_required(VERSION 3.20)
project(test-sync C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-sync src/main.c)
target_include_directories(test-sync PRIVATE src)
target_compile_options(test-sync PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-sync RUNTIME DESTINATION usr/bin/starry-test-suit)
