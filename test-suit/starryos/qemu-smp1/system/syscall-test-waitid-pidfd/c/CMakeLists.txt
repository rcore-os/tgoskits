cmake_minimum_required(VERSION 3.20)
project(test-waitid-pidfd C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-waitid-pidfd src/main.c)
target_include_directories(test-waitid-pidfd PRIVATE src ${CMAKE_CURRENT_SOURCE_DIR}/../../common)
target_compile_options(test-waitid-pidfd PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-waitid-pidfd RUNTIME DESTINATION usr/bin/starry-test-suit)
