cmake_minimum_required(VERSION 3.20)
project(test-utimensat C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-utimensat src/main.c)
target_include_directories(test-utimensat PRIVATE src)
target_compile_options(test-utimensat PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-utimensat RUNTIME DESTINATION usr/bin/starry-test-suit)
