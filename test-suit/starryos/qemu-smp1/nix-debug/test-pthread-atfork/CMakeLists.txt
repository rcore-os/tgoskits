cmake_minimum_required(VERSION 3.20)
project(test-pthread-atfork C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-pthread-atfork src/main.c)
target_include_directories(test-pthread-atfork PRIVATE src ${CMAKE_CURRENT_SOURCE_DIR}/../../system/common)
target_compile_options(test-pthread-atfork PRIVATE -Wall -Wextra -Werror)
target_link_options(test-pthread-atfork PRIVATE -static)
install(TARGETS test-pthread-atfork RUNTIME DESTINATION usr/bin/starry-test-suit)
