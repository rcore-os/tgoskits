cmake_minimum_required(VERSION 3.20)
project(test-mmap-private-fork-exit C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-mmap-private-fork-exit src/main.c)
target_include_directories(test-mmap-private-fork-exit PRIVATE src ${CMAKE_CURRENT_SOURCE_DIR}/../../system/common)
target_compile_options(test-mmap-private-fork-exit PRIVATE -Wall -Wextra -Werror)
target_link_options(test-mmap-private-fork-exit PRIVATE -static)
install(TARGETS test-mmap-private-fork-exit RUNTIME DESTINATION usr/bin/starry-test-suit)
