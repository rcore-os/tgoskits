cmake_minimum_required(VERSION 3.20)
project(test-pty-fork-echo C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-pty-fork-echo src/main.c)
target_include_directories(test-pty-fork-echo PRIVATE src ${CMAKE_CURRENT_SOURCE_DIR}/../../system/common)
target_compile_options(test-pty-fork-echo PRIVATE -Wall -Wextra -Werror)
target_link_options(test-pty-fork-echo PRIVATE -static)
install(TARGETS test-pty-fork-echo RUNTIME DESTINATION usr/bin/starry-test-suit)
