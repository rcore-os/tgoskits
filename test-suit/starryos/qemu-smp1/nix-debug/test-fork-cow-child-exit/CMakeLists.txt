cmake_minimum_required(VERSION 3.20)
project(test-fork-cow-child-exit C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-fork-cow-child-exit src/main.c)
target_include_directories(test-fork-cow-child-exit PRIVATE src ${CMAKE_CURRENT_SOURCE_DIR}/../../system/common)
target_compile_options(test-fork-cow-child-exit PRIVATE -Wall -Wextra -Werror)
target_link_options(test-fork-cow-child-exit PRIVATE -static)
install(TARGETS test-fork-cow-child-exit RUNTIME DESTINATION usr/bin/starry-test-suit)
