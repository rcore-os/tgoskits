cmake_minimum_required(VERSION 3.20)
project(test-mremap-realloc C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-mremap-realloc src/main.c)
target_include_directories(test-mremap-realloc PRIVATE src ${CMAKE_CURRENT_SOURCE_DIR}/../../system/common)
target_compile_options(test-mremap-realloc PRIVATE -Wall -Wextra -Werror)
target_link_options(test-mremap-realloc PRIVATE -static)
install(TARGETS test-mremap-realloc RUNTIME DESTINATION usr/bin/starry-test-suit)
