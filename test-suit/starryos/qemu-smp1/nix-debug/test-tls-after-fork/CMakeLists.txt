cmake_minimum_required(VERSION 3.20)
project(test-tls-after-fork C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-tls-after-fork src/main.c)
target_include_directories(test-tls-after-fork PRIVATE src ${CMAKE_CURRENT_SOURCE_DIR}/../../system/common)
target_compile_options(test-tls-after-fork PRIVATE -Wall -Wextra -Werror)
target_link_options(test-tls-after-fork PRIVATE -static)
install(TARGETS test-tls-after-fork RUNTIME DESTINATION usr/bin/starry-test-suit)
