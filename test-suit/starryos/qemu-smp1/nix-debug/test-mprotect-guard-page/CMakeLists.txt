cmake_minimum_required(VERSION 3.20)
project(test-mprotect-guard-page C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-mprotect-guard-page src/main.c)
target_include_directories(test-mprotect-guard-page PRIVATE src ${CMAKE_CURRENT_SOURCE_DIR}/../../system/common)
target_compile_options(test-mprotect-guard-page PRIVATE -Wall -Wextra -Werror)
target_link_options(test-mprotect-guard-page PRIVATE -static)
install(TARGETS test-mprotect-guard-page RUNTIME DESTINATION usr/bin/starry-test-suit)
