cmake_minimum_required(VERSION 3.20)
project(test-rapid-fork-exit-cow C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-rapid-fork-exit-cow src/main.c)
target_include_directories(test-rapid-fork-exit-cow PRIVATE src ${CMAKE_CURRENT_SOURCE_DIR}/../../system/common)
target_compile_options(test-rapid-fork-exit-cow PRIVATE -Wall -Wextra -Werror)
target_link_options(test-rapid-fork-exit-cow PRIVATE -static)
install(TARGETS test-rapid-fork-exit-cow RUNTIME DESTINATION usr/bin/starry-test-suit)
