cmake_minimum_required(VERSION 3.20)
project(test-nix-sandbox-probe C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-nix-sandbox-probe src/main.c)
target_include_directories(test-nix-sandbox-probe PRIVATE src ${CMAKE_CURRENT_SOURCE_DIR}/../../system/common)
target_compile_options(test-nix-sandbox-probe PRIVATE -Wall -Wextra -Werror)
target_link_options(test-nix-sandbox-probe PRIVATE -static)
install(TARGETS test-nix-sandbox-probe RUNTIME DESTINATION usr/bin/starry-test-suit)
