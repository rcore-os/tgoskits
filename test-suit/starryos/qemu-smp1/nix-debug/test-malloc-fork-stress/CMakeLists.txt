cmake_minimum_required(VERSION 3.20)
project(test-malloc-fork-stress C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-malloc-fork-stress src/main.c)
target_include_directories(test-malloc-fork-stress PRIVATE src ${CMAKE_CURRENT_SOURCE_DIR}/../../system/common)
target_compile_options(test-malloc-fork-stress PRIVATE -Wall -Wextra -Werror)
target_link_options(test-malloc-fork-stress PRIVATE -static)
install(TARGETS test-malloc-fork-stress RUNTIME DESTINATION usr/bin/starry-test-suit)
