cmake_minimum_required(VERSION 3.20)
project(test-madvise-dontneed-heap C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-madvise-dontneed-heap src/main.c)
target_include_directories(test-madvise-dontneed-heap PRIVATE src ${CMAKE_CURRENT_SOURCE_DIR}/../../system/common)
target_compile_options(test-madvise-dontneed-heap PRIVATE -Wall -Wextra -Werror)
target_link_options(test-madvise-dontneed-heap PRIVATE -static)
install(TARGETS test-madvise-dontneed-heap RUNTIME DESTINATION usr/bin/starry-test-suit)
