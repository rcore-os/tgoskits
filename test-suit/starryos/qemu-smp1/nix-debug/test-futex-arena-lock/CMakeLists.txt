cmake_minimum_required(VERSION 3.20)
project(test-futex-arena-lock C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-futex-arena-lock src/main.c)
target_include_directories(test-futex-arena-lock PRIVATE src ${CMAKE_CURRENT_SOURCE_DIR}/../../system/common)
target_compile_options(test-futex-arena-lock PRIVATE -Wall -Wextra -Werror)
target_link_options(test-futex-arena-lock PRIVATE -static)
install(TARGETS test-futex-arena-lock RUNTIME DESTINATION usr/bin/starry-test-suit)
