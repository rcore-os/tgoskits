cmake_minimum_required(VERSION 3.20)
project(test-concurrent-mmap C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-concurrent-mmap src/main.c)
target_include_directories(test-concurrent-mmap PRIVATE src)
target_compile_options(test-concurrent-mmap PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-concurrent-mmap RUNTIME DESTINATION usr/bin/starry-test-suit)
