cmake_minimum_required(VERSION 3.20)
project(test-syslog-sysinfo-uname C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-syslog-sysinfo-uname src/main.c)
target_include_directories(test-syslog-sysinfo-uname PRIVATE src)
target_compile_options(test-syslog-sysinfo-uname PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-syslog-sysinfo-uname RUNTIME DESTINATION usr/bin)
