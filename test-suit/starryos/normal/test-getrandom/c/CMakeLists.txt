cmake_minimum_required(VERSION 3.20)
project(test-getrandom C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-getrandom src/main.c)
target_include_directories(test-getrandom PRIVATE src)
target_compile_options(test-getrandom PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-getrandom RUNTIME DESTINATION usr/bin)
