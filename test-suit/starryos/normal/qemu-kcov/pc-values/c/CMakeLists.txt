cmake_minimum_required(VERSION 3.20)
project(test-kcov-pc-values C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-kcov-pc-values src/main.c)
target_include_directories(test-kcov-pc-values PRIVATE src)
target_compile_options(test-kcov-pc-values PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-kcov-pc-values RUNTIME DESTINATION usr/bin)
