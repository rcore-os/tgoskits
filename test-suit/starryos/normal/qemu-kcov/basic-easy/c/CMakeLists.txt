cmake_minimum_required(VERSION 3.20)
project(test-kcov-basic-easy C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-kcov-basic-easy src/main.c)
target_include_directories(test-kcov-basic-easy PRIVATE src)
target_compile_options(test-kcov-basic-easy PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-kcov-basic-easy RUNTIME DESTINATION usr/bin)
