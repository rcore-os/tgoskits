cmake_minimum_required(VERSION 3.20)
project(test-kcov-open C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-kcov-open src/main.c)
target_include_directories(test-kcov-open PRIVATE src)
target_compile_options(test-kcov-open PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-kcov-open RUNTIME DESTINATION usr/bin)
