cmake_minimum_required(VERSION 3.20)
project(test-kcov-disable-thread-check C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-kcov-disable-thread-check src/main.c)
target_include_directories(test-kcov-disable-thread-check PRIVATE src)
target_compile_options(test-kcov-disable-thread-check PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-kcov-disable-thread-check RUNTIME DESTINATION usr/bin)
