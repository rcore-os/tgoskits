cmake_minimum_required(VERSION 3.20)
project(test-kcov-guard C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-kcov-guard src/main.c)
target_include_directories(test-kcov-guard PRIVATE src)
target_compile_options(test-kcov-guard PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-kcov-guard RUNTIME DESTINATION usr/bin)
