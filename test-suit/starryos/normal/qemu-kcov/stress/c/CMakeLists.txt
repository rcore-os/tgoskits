cmake_minimum_required(VERSION 3.20)
project(test-kcov-stress C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-kcov-stress src/main.c)
target_include_directories(test-kcov-stress PRIVATE src)
target_compile_options(test-kcov-stress PRIVATE -Wall -Wextra -Werror)
target_link_libraries(test-kcov-stress PRIVATE -lpthread)
install(TARGETS test-kcov-stress RUNTIME DESTINATION usr/bin)
