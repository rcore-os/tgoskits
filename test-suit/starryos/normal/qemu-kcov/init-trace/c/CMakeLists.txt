cmake_minimum_required(VERSION 3.20)
project(test-kcov-init-trace C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-kcov-init-trace src/main.c)
target_include_directories(test-kcov-init-trace PRIVATE src)
target_compile_options(test-kcov-init-trace PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-kcov-init-trace RUNTIME DESTINATION usr/bin)
