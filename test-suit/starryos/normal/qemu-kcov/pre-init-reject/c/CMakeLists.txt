cmake_minimum_required(VERSION 3.20)
project(test-kcov-pre-init-reject C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-kcov-pre-init-reject src/main.c)
target_include_directories(test-kcov-pre-init-reject PRIVATE src)
target_compile_options(test-kcov-pre-init-reject PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-kcov-pre-init-reject RUNTIME DESTINATION usr/bin)
