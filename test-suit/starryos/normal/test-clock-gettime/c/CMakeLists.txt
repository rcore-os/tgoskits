cmake_minimum_required(VERSION 3.20)
project(test-clock-gettime C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-clock-gettime src/main.c)
target_include_directories(test-clock-gettime PRIVATE src)
target_compile_options(test-clock-gettime PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-clock-gettime RUNTIME DESTINATION usr/bin)
