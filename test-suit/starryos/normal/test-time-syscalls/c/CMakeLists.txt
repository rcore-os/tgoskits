cmake_minimum_required(VERSION 3.20)
project(test-time-syscalls C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-time-syscalls src/main.c)
target_include_directories(test-time-syscalls PRIVATE src)
target_compile_options(test-time-syscalls PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-time-syscalls RUNTIME DESTINATION usr/bin)
