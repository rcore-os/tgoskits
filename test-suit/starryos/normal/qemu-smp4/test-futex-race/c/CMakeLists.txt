cmake_minimum_required(VERSION 3.20)
project(test-futex-race C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-futex-race src/main.c)
target_compile_options(test-futex-race PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-futex-race RUNTIME DESTINATION usr/bin)
