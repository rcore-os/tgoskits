cmake_minimum_required(VERSION 3.20)
project(test-futex-robust C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-futex-robust src/main.c)
target_include_directories(test-futex-robust PRIVATE src)
target_compile_options(test-futex-robust PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-futex-robust RUNTIME DESTINATION usr/bin)
