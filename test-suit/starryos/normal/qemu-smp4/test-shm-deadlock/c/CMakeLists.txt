cmake_minimum_required(VERSION 3.20)
project(test-shm-deadlock C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-shm-deadlock src/main.c)
target_include_directories(test-shm-deadlock PRIVATE src)
target_compile_options(test-shm-deadlock PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-shm-deadlock RUNTIME DESTINATION usr/bin)
