cmake_minimum_required(VERSION 3.20)
project(test-credentials C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-credentials src/main.c)
target_include_directories(test-credentials PRIVATE src)
target_compile_options(test-credentials PRIVATE -Wall -Wextra)
install(TARGETS test-credentials RUNTIME DESTINATION usr/bin)
