cmake_minimum_required(VERSION 3.20)
project(test-kcov-concurrent C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-kcov-concurrent src/main.c)
target_include_directories(test-kcov-concurrent PRIVATE src)
target_compile_options(test-kcov-concurrent PRIVATE -Wall -Wextra -Werror)
target_link_libraries(test-kcov-concurrent PRIVATE -lpthread)
install(TARGETS test-kcov-concurrent RUNTIME DESTINATION usr/bin)
