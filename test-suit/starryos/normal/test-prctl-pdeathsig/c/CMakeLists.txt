cmake_minimum_required(VERSION 3.20)
project(test-prctl-pdeathsig C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-prctl-pdeathsig src/main.c)
target_include_directories(test-prctl-pdeathsig PRIVATE src)
target_compile_options(test-prctl-pdeathsig PRIVATE -Wall -Wextra)
install(TARGETS test-prctl-pdeathsig RUNTIME DESTINATION usr/bin)
