cmake_minimum_required(VERSION 3.20)
project(test-netlink-rtnetlink C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-netlink-rtnetlink src/main.c)
target_include_directories(test-netlink-rtnetlink PRIVATE src)
target_compile_options(test-netlink-rtnetlink PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-netlink-rtnetlink RUNTIME DESTINATION usr/bin)
