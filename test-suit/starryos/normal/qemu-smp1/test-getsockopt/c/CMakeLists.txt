cmake_minimum_required(VERSION 3.20)
project(test-getsockopt C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-getsockopt src/main.c)
target_include_directories(test-getsockopt PRIVATE src)
target_compile_options(test-getsockopt PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-getsockopt RUNTIME DESTINATION usr/bin)
