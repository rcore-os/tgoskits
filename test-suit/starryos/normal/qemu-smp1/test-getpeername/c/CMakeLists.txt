cmake_minimum_required(VERSION 3.20)
project(test-getpeername C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-getpeername src/main.c)
target_include_directories(test-getpeername PRIVATE src)
target_compile_options(test-getpeername PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-getpeername RUNTIME DESTINATION usr/bin)
