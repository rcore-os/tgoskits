cmake_minimum_required(VERSION 3.20)
project(test-drm-version C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-drm-version src/main.c)
target_include_directories(test-drm-version PRIVATE src)
target_compile_options(test-drm-version PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-drm-version RUNTIME DESTINATION usr/bin)
