cmake_minimum_required(VERSION 3.20)
project(test-ftruncate C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-ftruncate src/main.c)
target_include_directories(test-ftruncate PRIVATE src)
target_compile_options(test-ftruncate PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-ftruncate RUNTIME DESTINATION usr/bin/starry-test-suit)
