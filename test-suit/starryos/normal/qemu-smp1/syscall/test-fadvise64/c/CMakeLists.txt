cmake_minimum_required(VERSION 3.20)
project(test-fadvise64 C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-fadvise64 src/main.c)
target_include_directories(test-fadvise64 PRIVATE src)
target_compile_options(test-fadvise64 PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-fadvise64 RUNTIME DESTINATION usr/bin/starry-test-suit)
