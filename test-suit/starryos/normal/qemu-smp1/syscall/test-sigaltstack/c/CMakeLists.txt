cmake_minimum_required(VERSION 3.20)
project(test-sigaltstack C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-sigaltstack src/main.c)
target_include_directories(test-sigaltstack PRIVATE src)
target_compile_options(test-sigaltstack PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-sigaltstack RUNTIME DESTINATION usr/bin)
