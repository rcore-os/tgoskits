cmake_minimum_required(VERSION 3.20)
project(test-faccessat C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-faccessat src/main.c)
target_include_directories(test-faccessat PRIVATE src)
target_compile_options(test-faccessat PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-faccessat RUNTIME DESTINATION usr/bin)
