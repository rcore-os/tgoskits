cmake_minimum_required(VERSION 3.20)
project(test-fchown C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-fchown src/main.c)
target_include_directories(test-fchown PRIVATE src)
target_compile_options(test-fchown PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-fchown RUNTIME DESTINATION usr/bin)
