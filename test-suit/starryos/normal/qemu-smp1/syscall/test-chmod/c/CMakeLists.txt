cmake_minimum_required(VERSION 3.20)
project(test-chmod C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-chmod src/main.c)
target_include_directories(test-chmod PRIVATE src)
target_compile_options(test-chmod PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-chmod RUNTIME DESTINATION usr/bin)
