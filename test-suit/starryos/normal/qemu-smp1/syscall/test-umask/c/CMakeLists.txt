cmake_minimum_required(VERSION 3.20)
project(test-umask C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-umask src/main.c)
target_include_directories(test-umask PRIVATE src)
target_compile_options(test-umask PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-umask RUNTIME DESTINATION usr/bin)
