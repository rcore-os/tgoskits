cmake_minimum_required(VERSION 3.20)
project(test-signalfd4 C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-signalfd4 src/main.c)
target_include_directories(test-signalfd4 PRIVATE src)
target_compile_options(test-signalfd4 PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-signalfd4 RUNTIME DESTINATION usr/bin/starry-test-suit)
