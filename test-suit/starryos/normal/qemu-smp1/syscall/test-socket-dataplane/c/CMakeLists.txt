cmake_minimum_required(VERSION 3.20)
project(test-socket-dataplane C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-socket-dataplane src/main.c)
target_include_directories(test-socket-dataplane PRIVATE src)
target_compile_options(test-socket-dataplane PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-socket-dataplane RUNTIME DESTINATION usr/bin/starry-test-suit)
