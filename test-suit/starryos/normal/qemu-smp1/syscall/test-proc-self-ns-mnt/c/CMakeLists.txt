cmake_minimum_required(VERSION 3.20)
project(test-proc-self-ns-mnt C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-proc-self-ns-mnt src/main.c)
target_include_directories(test-proc-self-ns-mnt PRIVATE src ${CMAKE_CURRENT_SOURCE_DIR}/../../common)

install(TARGETS test-proc-self-ns-mnt RUNTIME DESTINATION usr/bin/starry-test-suit)
