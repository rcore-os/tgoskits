cmake_minimum_required(VERSION 3.20)
project(test-fchownat C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-fchownat src/main.c)
target_include_directories(test-fchownat PRIVATE src)
target_compile_options(test-fchownat PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-fchownat RUNTIME DESTINATION usr/bin)
