cmake_minimum_required(VERSION 3.20)
project(test-stat-family C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-stat-family src/main.c)
target_include_directories(test-stat-family PRIVATE src)
target_compile_options(test-stat-family PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-stat-family RUNTIME DESTINATION usr/bin)
