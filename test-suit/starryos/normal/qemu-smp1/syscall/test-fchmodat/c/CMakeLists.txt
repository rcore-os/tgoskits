cmake_minimum_required(VERSION 3.20)
project(test-fchmodat C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-fchmodat src/main.c)
target_include_directories(test-fchmodat PRIVATE src)
target_compile_options(test-fchmodat PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-fchmodat RUNTIME DESTINATION usr/bin)
