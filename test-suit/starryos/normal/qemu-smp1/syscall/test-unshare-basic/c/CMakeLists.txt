cmake_minimum_required(VERSION 3.20)
project(test-unshare-basic C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-unshare-basic src/main.c)
target_include_directories(test-unshare-basic PRIVATE src ${CMAKE_CURRENT_SOURCE_DIR}/../../common)
target_compile_options(test-unshare-basic PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-unshare-basic RUNTIME DESTINATION usr/bin/starry-test-suit)
