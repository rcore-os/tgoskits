cmake_minimum_required(VERSION 3.20)
project(test-fchmod C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-fchmod src/main.c)
target_include_directories(test-fchmod PRIVATE src)
target_compile_options(test-fchmod PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-fchmod RUNTIME DESTINATION usr/bin)
