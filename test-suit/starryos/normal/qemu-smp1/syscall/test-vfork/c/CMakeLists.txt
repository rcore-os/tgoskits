cmake_minimum_required(VERSION 3.20)
project(test-vfork C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-vfork src/main.c)
target_include_directories(test-vfork PRIVATE src)
target_compile_options(test-vfork PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-vfork RUNTIME DESTINATION usr/bin)
