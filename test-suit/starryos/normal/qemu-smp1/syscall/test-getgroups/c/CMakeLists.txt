cmake_minimum_required(VERSION 3.20)
project(test-getgroups C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-getgroups src/main.c)
target_include_directories(test-getgroups PRIVATE src)
target_compile_options(test-getgroups PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-getgroups RUNTIME DESTINATION usr/bin)
