cmake_minimum_required(VERSION 3.20)
project(test-chown C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-chown src/main.c)
target_include_directories(test-chown PRIVATE src)
target_compile_options(test-chown PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-chown RUNTIME DESTINATION usr/bin)
