cmake_minimum_required(VERSION 3.20)
project(test-drm-modeset C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-drm-modeset src/main.c)
target_include_directories(test-drm-modeset PRIVATE src)
target_compile_options(test-drm-modeset PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-drm-modeset RUNTIME DESTINATION usr/bin)
