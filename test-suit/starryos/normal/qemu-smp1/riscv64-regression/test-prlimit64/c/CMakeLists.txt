cmake_minimum_required(VERSION 3.20)
project(test-prlimit64 C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-prlimit64 src/main.c)
target_include_directories(test-prlimit64 PRIVATE src)
target_compile_options(test-prlimit64 PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-prlimit64 RUNTIME DESTINATION usr/bin)
