cmake_minimum_required(VERSION 3.20)
project(test-sockopt C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-sockopt src/main.c)
target_include_directories(test-sockopt PRIVATE src)
target_compile_options(test-sockopt PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-sockopt RUNTIME DESTINATION usr/bin)
