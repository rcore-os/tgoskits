cmake_minimum_required(VERSION 3.20)
project(test-drm-perbuf-dumb C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-drm-perbuf-dumb src/main.c)
target_include_directories(test-drm-perbuf-dumb PRIVATE src)
target_compile_options(test-drm-perbuf-dumb PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-drm-perbuf-dumb RUNTIME DESTINATION usr/bin)
