cmake_minimum_required(VERSION 3.20)
project(test-memfd-write-busy C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-memfd-write-busy src/main.c)
target_include_directories(test-memfd-write-busy PRIVATE src)
target_compile_options(test-memfd-write-busy PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-memfd-write-busy RUNTIME DESTINATION usr/bin)
