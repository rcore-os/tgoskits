cmake_minimum_required(VERSION 3.20)
project(test-clock-getres C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-clock-getres src/main.c)
target_include_directories(test-clock-getres PRIVATE src)
target_compile_options(test-clock-getres PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-clock-getres RUNTIME DESTINATION usr/bin)
