cmake_minimum_required(VERSION 3.20)
project(test-setsockopt C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-setsockopt src/main.c)
target_include_directories(test-setsockopt PRIVATE src)
target_compile_options(test-setsockopt PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-setsockopt RUNTIME DESTINATION usr/bin)
