cmake_minimum_required(VERSION 3.20)
project(bug-kill-zombie-esrch C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(bug-kill-zombie-esrch src/main.c)
target_include_directories(bug-kill-zombie-esrch PRIVATE src)
target_compile_options(bug-kill-zombie-esrch PRIVATE -Wall -Wextra -Werror)
install(TARGETS bug-kill-zombie-esrch RUNTIME DESTINATION usr/bin)
