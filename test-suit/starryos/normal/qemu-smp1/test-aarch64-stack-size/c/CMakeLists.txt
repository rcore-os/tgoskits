cmake_minimum_required(VERSION 3.20)
project(test-aarch64-stack-size C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-aarch64-stack-size src/main.c)
target_include_directories(test-aarch64-stack-size PRIVATE src)
target_compile_options(test-aarch64-stack-size PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-aarch64-stack-size RUNTIME DESTINATION usr/bin)
