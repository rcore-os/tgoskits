cmake_minimum_required(VERSION 3.20)
project(test-fallocate C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-fallocate src/main.c)
target_include_directories(test-fallocate PRIVATE src)
target_compile_options(test-fallocate PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-fallocate RUNTIME DESTINATION usr/bin)
