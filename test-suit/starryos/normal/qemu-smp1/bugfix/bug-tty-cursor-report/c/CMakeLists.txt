cmake_minimum_required(VERSION 3.20)
project(bug-tty-cursor-report C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(bug-tty-cursor-report src/main.c)
target_include_directories(bug-tty-cursor-report PRIVATE src)
target_compile_options(bug-tty-cursor-report PRIVATE -Wall -Wextra -Werror)
target_compile_definitions(bug-tty-cursor-report PRIVATE _POSIX_C_SOURCE=200809L)
install(TARGETS bug-tty-cursor-report RUNTIME DESTINATION usr/bin)
