cmake_minimum_required(VERSION 3.20)
project(bug-alarm-pause C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS ON)
add_executable(bug-alarm-pause src/main.c)
target_compile_options(bug-alarm-pause PRIVATE -Wall -Wextra -Werror)
install(TARGETS bug-alarm-pause RUNTIME DESTINATION usr/bin)
