cmake_minimum_required(VERSION 3.20)
project(test-sysfs-class C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-sysfs-class src/main.c)
target_include_directories(test-sysfs-class PRIVATE src)
target_compile_options(test-sysfs-class PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-sysfs-class RUNTIME DESTINATION usr/bin)
