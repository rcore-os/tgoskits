cmake_minimum_required(VERSION 3.20)
project(starry-udp-test C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(starry-udp-test src/main.c)
target_compile_options(starry-udp-test PRIVATE -Wall -Wextra -Werror)
install(TARGETS starry-udp-test RUNTIME DESTINATION usr/bin)
