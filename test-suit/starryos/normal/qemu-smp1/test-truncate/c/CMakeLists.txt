cmake_minimum_required(VERSION 3.20)
project(test-truncate C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-truncate src/main.c)
target_include_directories(test-truncate PRIVATE src)
target_compile_options(test-truncate PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-truncate RUNTIME DESTINATION usr/bin)
