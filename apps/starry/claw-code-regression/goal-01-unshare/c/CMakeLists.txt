cmake_minimum_required(VERSION 3.20)
project(test-claw-code-goal-01-unshare C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-claw-code-goal-01-unshare src/main.c)
target_include_directories(test-claw-code-goal-01-unshare PRIVATE src)
target_compile_options(test-claw-code-goal-01-unshare PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-claw-code-goal-01-unshare RUNTIME DESTINATION usr/bin)
